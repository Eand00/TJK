-- creation of the table card_deck

CREATE TABLE card_deck (
    ID_CARD INT,
    ID_DECK INT,
    PRIMARY KEY (ID_CARD, ID_DECK),
    FOREIGN KEY (ID_CARD) REFERENCES PokemonCard(ID_CARD)
        ON DELETE CASCADE
        ON UPDATE CASCADE,
    FOREIGN KEY (ID_DECK) REFERENCES deck(ID_DECK)
        ON DELETE CASCADE
        ON UPDATE CASCADE
);
